// Lightbox const lightbox = document.getElementById("lightbox"); const lightboxImg = document.getElementById("lightbox-img"); const closeBtn = document.querySelector(".lightbox-close"); document.querySelectorAll(".gallery-img").forEach(img => { img.addEventListener("click", () => { lightboxImg.src = img.src; lightboxImg.alt = img.alt || "Gallery image"; lightbox.classList.add("open"); lightbox.setAttribute("aria-hidden", "false"); }); }); function closeLightbox() { lightbox.classList.remove("open"); lightbox.setAttribute("aria-hidden", "true"); lightboxImg.src = ""; } closeBtn.addEventListener("click", closeLightbox); lightbox.addEventListener("click", (e) => { if (e.target === lightbox) closeLightbox(); }); document.addEventListener("keydown", (e) => { if (e.key === "Escape") closeLightbox(); });
